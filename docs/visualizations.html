Importación de librerías

import pandas as pd
import altair as alt

alt.theme.enable('fivethirtyeight')
ThemeRegistry.enable('fivethirtyeight')

df_final = pd.read_csv("df_final.csv")

¿Cómo cambian los niveles de ruido a lo largo de un mes completo?

Preparación de datos

# Clasificamos las horas
df_hora_promedio = df_final.groupby('Hora')['Medición (dB)'].mean().reset_index()

orden_franjas = ['Madrugada', 'Mañana', 'Tarde', 'Noche']

df_fondo = pd.DataFrame({
    'inicio': [0, 6, 12, 18],
    'fin': [6, 12, 18, 24],
    'Franja': ['Madrugada', 'Mañana', 'Tarde', 'Noche']
})

Fondo por franjas horarias

# Colores diferentes para diferenciar los horarios 
fondo_sombras = alt.Chart(df_fondo).mark_rect(opacity=0.1).encode(
    x='inicio:Q',
    x2='fin:Q',
    color=alt.Color(
        'Franja:N',
        scale=alt.Scale(domain=orden_franjas, scheme='category10'),
        legend=alt.Legend(title="Fase", orient='right', labelFontSize=10)
    )
)

Línea de tendencia

# Linea de tendencia en los diferentes horas 
linea_tendencia = alt.Chart(df_hora_promedio).mark_line(
    color='#1a1a1a',
    strokeWidth=2.5,
    interpolate='monotone'
).encode(
    x=alt.X('Hora:Q', title=None, scale=alt.Scale(domain=[0, 23])),
    y=alt.Y(
        'Medición (dB):Q',
        title='dB Promedio',
        scale=alt.Scale(zero=False, domain=[60, 78])
    )
)

Puntos

# Puntos de la medicion horaria 
puntos_detalle = alt.Chart(df_hora_promedio).mark_circle(size=50).encode(
    x='Hora:Q',
    y='Medición (dB):Q',
    tooltip=[
        alt.Tooltip('Hora:Q', title='Hora'),
        alt.Tooltip('Medición (dB):Q', title='dB promedio')
    ],
    color=alt.Color(
        'Medición (dB):Q',
        scale=alt.Scale(scheme='magma'),
        legend=None
    )
)

Gráfico combinado

# Union de graficos
grafico_linea = (fondo_sombras + linea_tendencia + puntos_detalle).properties(
    width=575,
    height=200,
    title=alt.TitleParams("Ritmo y Densidad Acústica", fontSize=16)
)

Mapa de calor

# Transformamos las fechas a su typo correspondiente
df_final['Fecha'] = pd.to_datetime(df_final['Fecha'], errors='coerce')
df_final['Fecha_Etiqueta'] = df_final['Fecha'].dt.strftime('%d-%m')

# Creamos el mapa de calor
mapa_calor = alt.Chart(df_final).mark_rect().encode(
    x=alt.X('Hora:O', title='Hora (0-23)', axis=alt.Axis(labelAngle=0)),
    y=alt.Y('Fecha_Etiqueta:O', title='Día/Mes'),
    color=alt.Color('Medición (dB):Q', title='dB', scale=alt.Scale(scheme='magma')),
    tooltip=['Fecha_Etiqueta', 'Hora', 'Medición (dB)']
).properties(
    width=575,
    height=280
)

Dashbord final

# Cnfiguramos label y title 
dashboard_final = (grafico_linea & mapa_calor).configure_axis(
    labelFontSize=9,
    titleFontSize=11

dashboard_final

Conclusión

El análisis del ritmo y densidad acústica evidencia un patrón diario claramente estructurado. A partir de las 6:00 a.m. el nivel de ruido se incrementa de forma abrupta, manteniéndose por encima de los 70 dB durante aproximadamente 16 horas continuas, lo que indica una exposición prolongada a niveles de ruido.

El mapa de calor muestra que el único periodo de disminución significativa del ruido ocurre entre la 1:00 a.m. y las 5:00 a.m., considerado como la principal ventana de descanso acústico. No obstante, se observan eventos aislados de ruido en horas nocturnas, lo que sugiere interrupciones puntuales del silencio.

La zona presenta una exposición acústica constante durante el día, con un único periodo reducido de baja intensidad en la madrugada, reflejando un entorno urbano de alta carga sonora sostenida.

El análisis comparativo de los diferentes estados climáticos evidencia que el ruido en la zona no depende de las condiciones atmosféricas. Incluso en escenarios como la niebla moderada, los niveles promedio superan los 70 dB, alcanzando valores similares o superiores a los observados en días despejados o soleados.

Aunque en condiciones de lluvia ligera se observa una mayor dispersión hacia valores más bajos, los promedios de todas las categorías climáticas se mantienen cercanos al umbral de riesgo.

La contaminación acústica se mantiene constante independientemente del estado del clima, lo que confirma que su origen está asociado principalmente a la actividad urbana y no a factores naturales.

Mapa de Calor

# Asegurar formato de fecha
df_final['Fecha'] = pd.to_datetime(df_final['Fecha'])

# Mapa de calor semanal de ruido
heatmap_semanal = alt.Chart(df_final).mark_rect().encode(
    x=alt.X('Hora:O', title='Hora del Día', axis=alt.Axis(labelAngle=0)),
    y=alt.Y(
        'day(Fecha):O',
        title='Día de la Semana',
        sort=['Monday','Tuesday','Wednesday','Thursday','Friday','Saturday','Sunday']
    ),
    color=alt.Color(
        'mean(Medición (dB)):Q',
        title='Ruido Promedio (dB)',
        scale=alt.Scale(scheme='magma', domain=[60, 75])
    ),
    tooltip=[
        alt.Tooltip('day(Fecha):N', title='Día'),
        alt.Tooltip('Hora:O', title='Hora'),
        alt.Tooltip('mean(Medición (dB)):Q', format='.1f', title='Promedio dB'),
        alt.Tooltip('count():Q', title='Muestras')
    ]
).properties(
    width=800,
    height=200,
    title="Patrón Semanal de Contaminación Acústica"
)
# Área con gradiente
area_elegante = alt.Chart(df_final).mark_area(
    line={'color':'#2c3e50'},
    color=alt.Gradient(
        gradient='linear',
        stops=[
            alt.GradientStop(color='#3498db', offset=0),
            alt.GradientStop(color='#e74c3c', offset=1)
        ],
        x1=1, x2=1, y1=1, y2=0
    ),
    interpolate='monotone',
    opacity=0.6
).encode(
    x=alt.X('Hora:O', title='Hora del Día', axis=alt.Axis(labelAngle=0, grid=False)),
    y=alt.Y('mean(Medición (dB)):Q', title='Ruido Promedio (dB)', scale=alt.Scale(domain=[60, 75])),
    tooltip=['Hora:O', alt.Tooltip('mean(Medición (dB)):Q', format='.1f')]
)

# Puntos
puntos = area_elegante.mark_point(color='#2c3e50', size=40)

# Línea de umbral
linea_salud = alt.Chart(pd.DataFrame({'y': [70]})).mark_rule(
    color='#c0392b',
    strokeDash=[4, 4]
).encode(y='y:Q')

# Gráfico final
grafica_pro = area_elegante + puntos + linea_salud
# Unión vertical de gráficos
grafica_1 = grafica_pro & heatmap_semanal
# Unión verti
grafica_1.configure_axis(
    labelFontSize=11,
    titleFontSize=12,
    labelColor='#7f8c8d',
    titleColor='#2c3e50',
    grid=False
).configure_view(
    stroke=None
)

Conclusión

El análisis del patrón semanal del ruido evidencia una relación directa con la actividad humana. Durante los días laborales, el nivel de ruido supera los 70 dB desde aproximadamente las 6:00 a.m., manteniéndose elevado a lo largo del día debido al tráfico y las actividades comerciales.

En contraste, durante el fin de semana se observa un cambio en el comportamiento, con un desplazamiento del ruido hacia horas más nocturnas, especialmente entre la noche del viernes y la madrugada del sábado, lo que refleja dinámicas asociadas al ocio.

La contaminación acústica presenta un patrón claramente estructurado por los horarios de la ciudad, con mayor actividad en días laborables y un comportamiento más nocturno durante el fin de semana.

Visualización de la Densidad de Eventos: Relación entre Temperatura Ambiente y Contaminación Sonora

# 1. El cuerpo central: Mapa de densidad (Heatmap) con Tooltip
main = alt.Chart(df_final).mark_rect().encode(
    x=alt.X('Temperatura (°C):Q', bin=alt.Bin(maxbins=30), title='Temperatura (°C)'),
    y=alt.Y('Medición (dB):Q', bin=alt.Bin(maxbins=30), title='Ruido (dB)'),
    color=alt.Color('count():Q', scale=alt.Scale(scheme='viridis'), title='Densidad'),
    tooltip=[
        alt.Tooltip('count():Q', title='Registros en este bloque'),
        alt.Tooltip('mean(Medición (dB)):Q', format='.1f', title='Ruido Promedio'),
        alt.Tooltip('mean(Temperatura (°C)):Q', format='.1f', title='Temp Promedio')
    ]
).properties(width=600, height=400)

# 2. Histograma Superior con Tooltip (Valores de Temperatura)
top = alt.Chart(df_final).mark_bar(color='#440154').encode(
    x=alt.X('Temperatura (°C):Q', bin=alt.Bin(maxbins=30), axis=None),
    y=alt.Y('count():Q', title=None, axis=None),
    # Añadimos tooltip aquí
    tooltip=[
        alt.Tooltip('count():Q', title='Total de registros'),
        alt.Tooltip('Temperatura (°C):Q', bin=alt.Bin(maxbins=30), title='Rango de Temp')
    ]
).properties(width=600, height=60)

# 3. Histograma Lateral con Tooltip (Valores de Ruido)
right = alt.Chart(df_final).mark_bar(color='#fde725').encode(
    y=alt.Y('Medición (dB):Q', bin=alt.Bin(maxbins=30), axis=None),
    x=alt.X('count():Q', title=None, axis=None),
    # Añadimos tooltip aquí
    tooltip=[
        alt.Tooltip('count():Q', title='Total de registros'),
        alt.Tooltip('Medición (dB):Q', bin=alt.Bin(maxbins=30), title='Rango de Ruido')
    ]
).properties(width=80, height=400)

# 4. Ensamble Final
correlacion_grafica = alt.vconcat(
    top,
    alt.hconcat(main, right),
    spacing=5
).configure_view(
    stroke=None
).properties(
    title={
        "text": "Análisis de Correlación Interactivo",
        "anchor": "start",
        "fontSize": 20
    }
)

correlacion_grafica

Conclusión

El análisis de la relación entre temperatura y ruido mediante el mapa de calor muestra que la mayor concentración de registros se presenta alrededor de los 17 °C, manteniéndose el nivel de ruido cercano a los 70 dB. Los histogramas complementarios confirman que, a pesar de las variaciones de temperatura a lo largo del día, el ruido urbano se mantiene constante y elevado.

¿De qué manera afectan las lluvias intensas a los niveles acústicos de la zona?

Segmentación de Precipitaciones

def cat_precip(p):
    if p == 0:      return 'Sin lluvia'
    elif p < 0.5:   return 'Ligera'
    elif p < 1.0:   return 'Moderada'
    else:           return 'Fuerte'

df_final = df_final.copy()  # evita el SettingWithCopyWarning
df_final['lluvia_cat'] = df_final['Precipitaciones (mm)'].apply(cat_precip)

# Verificar
print(df_final['lluvia_cat'].value_counts())
print("Nulls:", df_final['lluvia_cat'].isna().sum())

Conclusión

Los resultados muestran que la lluvia no reduce el ruido ambiental. Independientemente del nivel de precipitación, el ruido se mantiene cercano a los 70 dB, lo que evidencia una alta persistencia del nivel sonoro urbano. Además, la lluvia tiende a disminuir los momentos de menor ruido, generando un ambiente más constante debido al impacto del agua y al contacto de los vehículos con el pavimento mojado. La contaminación acústica en la zona analizada está principalmente asociada al tráfico y la actividad humana, por lo que no depende de las condiciones climáticas.
