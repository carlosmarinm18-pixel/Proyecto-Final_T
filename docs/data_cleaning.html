Preparación de datos (Clima)

# Preparar el dataset de Clima
# Procedemos a darle el typo correspondiente y a cambiar las comas por puntos
df_clima['Temperatura (°C)'] = pd.to_numeric(
    df_clima['Temperatura (°C)'].astype(str).str.replace(',', '.'),
    errors='coerce'
)

df_clima['Precipitaciones (mm)'] = pd.to_numeric(
    df_clima['Precipitaciones (mm)'].astype(str).str.replace(',', '.'),
    errors='coerce'
)

df_clima['Fecha'] = pd.to_datetime(df_clima['Fecha'], errors='coerce')

# Rellenar los valores NaN con esta media
